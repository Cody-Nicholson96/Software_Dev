# Math

```c++
// ---------- ARITHMETIC ----------
// The arithmetic operators are +, -, *, /, %, ++, --
 
cout << "5 + 2 = " << 5+2 << endl;
cout << "5 - 2 = " << 5-2 << endl;
cout << "5 * 2 = " << 5*2 << endl;
cout << "5 / 2 = " << 5/2 << endl;
cout << "5 % 2 = " << 5%2 << endl;
 
int five = 5;
cout << "5++ = " << five++ << endl;
cout << "++5 = " << ++five << endl;
cout << "5-- = " << five-- << endl;
cout << "--5 = " << --five << endl;
 
// Shorthand assignment operators
// a += b == a = a + b
// There is also -=, *=, /=, %=
 
// Order of Operation states * and / is performed before + and -
 
cout << "1 + 2 - 3 * 2 = " << 1 + 2 - 3 * 2 << endl;
cout << "(1 + 2 - 3) * 2 = " << (1 + 2 - 3) * 2 << endl;
```
